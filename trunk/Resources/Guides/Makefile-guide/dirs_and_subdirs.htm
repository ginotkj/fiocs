FTPDIR = ftp-dir
TFTPDIR = tftp-dir
DNSDIR = dns-dir
COMDIR = common-dir
SUBDIRS = $(COMDIR) $(FTPDIR) $(TFTPDIR) $(DNSDIR)
# Default Target
all:
@echo
@echo "#######################################"
@echo "### BUILDING ALL TARGETS ###"
@echo "#######################################"
@echo
for i in $(SUBDIRS) ; do \
( cd $$i ; make ) ; \
done
# Rules to build individual targets
libs:
@cd $(COMDIR) ; make
ftp:
@cd $(FTPDIR) ; make
tftp:
@cd $(TFTPDIR) ; make
dnsresolver:
@cd $(DNSDIR) ; make
clean:
rm -f *~
for i in $(SUBDIRS) ; do \
( cd $$i ; make clean) ; \
done
install:
for i in $(SUBDIRS) ; do \
( cd $$i ; make install); \
done
